```python
#Exercism Code:
class HighScores:
    def __init__(self, scores):
        self.scores = scores

    def personal_best(self):
        return max(self.scores)

    def latest(self):
        return self.scores[-1]

    def personal_top_three(self):
        working_list = self.scores.copy()

        max_scores = []
        if len(working_list) > 3:       
            for score in range(3):
                best = max(working_list)
                working_list.remove(best)
                max_scores.append(best)
            return max_scores
        else: 
            return sorted(self.scores, reverse=True)
```
