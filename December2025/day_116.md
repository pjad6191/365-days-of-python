def triplets_with_sum(number):
    combos = []
    for a in range(1, number//3):
        for b in range(a+1, number//2):
            c = number - a - b

def triplets_with_sum(number):
    
    combos = []

    for a in range(1, number // 3):
        numerator = number * (number - 2 * a)
        denominator = 2 * (number - a)

        # b must be an integer
        if denominator == 0:
            continue

        if numerator % denominator != 0:
            continue  

        b = numerator // denominator
        c = number - a - b

        # enforce ordering and positivity
        if a < b < c and a > 0 and b > 0 and c > 0:
            if a * a + b * b == c * c: 
                combos.append([a, b, c])
    return combos
