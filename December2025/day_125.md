```python
#Exercism
class Matrix:
    def __init__(self, matrix_string):
        self.matrix_string = matrix_string

        # replace spaces with commas to delineate numbers
        new_string = self.matrix_string.replace(" ", ",")

        # create the matrix structure 
        self.matrix = []
        rows = new_string.split("\n")
        for line in rows:
            nums = (line.split(","))
            numeric = []
            for char in nums:
                numeric.append(int(char))
            self.matrix.append(numeric)
        

    def row(self, index):
        return self.matrix[index-1]

    
    def column(self, index):
        column = []
        for row in self.matrix:
            column.append(row[index-1])
        return column
```
