#Exercism
def recite(start, take=1):
    numbers = {10:"Ten", 9:"Nine", 8:"Eight", 7:"Seven", 6:"Six", 5:"Five", 4:"Four", 3:"Three", 2:"Two", 1:"One"}
    third_line = "And if one green bottle should accidentally fall,"
    space = ""
    song = []
    for verse in range(start, start-take, -1):
        if verse == 1:
            first_lines = "One green bottle hanging on the wall,"
            last_line = "There'll be no green bottles hanging on the wall."
        elif verse == 2:
            first_lines = "Two green bottles hanging on the wall,"
            last_line = "There'll be one green bottle hanging on the wall." 
        else:
            first_lines = f"{numbers[verse]} green bottles hanging on the wall,"
            last_line = f"There'll be {numbers[verse-1].lower()} green bottles hanging on the wall." 
        song_verse = [first_lines, first_lines, third_line, last_line]
        song += song_verse 
        if verse != start - take + 1:
            song.append("")
    return song
