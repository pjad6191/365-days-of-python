def convert(input_grid):
    numbers = {
        "0" : [" _ ", "| |", "|_|", "   "], 
        "1" : ["   ", "  |", "  |", "   "],
        "2" : [" _ ", " _|", "|_ ", "   "],
        "3" : [" _ ", " _|", " _|", "   "],
        "4" : ["   ", "|_|", "  |", "   "],
        "5" : [" _ ", "|_ ", " _|", "   "],
        "6" : [" _ ", "|_ ", "|_|", "   "],
        "7" : [" _ ", "  |", "  |", "   "],
        "8" : [" _ ", "|_|", "|_|", "   "],
        "9" : [" _ ", "|_|", " _|", "   "]}

    num_rows = len(input_grid)
    if num_rows == 0:
        return ""

    # When the rows aren't multiples of 4
    if num_rows % 4 != 0:
        raise ValueError("Number of input lines is not a multiple of four")

    num_cols = len(input_grid[0])

    # When the columns aren't multiples of 3
    for row in input_grid:
        if len(row) != num_cols:
            raise ValueError("Number of input columns is not a multiple of three")
        if num_cols % 3 != 0:
            raise ValueError("Number of input columns is not a multiple of three")

    lines_result = []   

    row_start = 0
    while row_start < num_rows:
        block_rows = input_grid[row_start:row_start + 4]

        line_digits = ""   

        col_start = 0
        while col_start < num_cols:
            number = []
            for r in range(4):
                row_segment = block_rows[r][col_start:col_start + 3]
                number.append(row_segment)

            digit = "?"
            for key, pattern in numbers.items():
                if number == pattern:
                    digit = key
                    break

            line_digits += digit
            col_start += 3

        lines_result.append(line_digits)
        row_start += 4

   
    return ",".join(lines_result)
