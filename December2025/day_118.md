#Exercism Code:
class School:
    def __init__(self):
        self.students = {}
        self.successfully_added = []
        

    def add_student(self, name, grade):
        if any(name in student_list for student_list in self.students.values()):
            self.successfully_added.append(False)
            return
        
        if grade in self.students:
            self.students[grade].append(name)
            self.students[grade].sort()
        else:
            self.students[grade] = [name]
        
        self.successfully_added.append(True)

            
    def roster(self):
        all_students = []
        sorted_grades = sorted(self.students.items())
        for grade, students in sorted_grades:
            all_students.extend(students)
        return all_students


    def grade(self, grade_number):
        if grade_number in self.students:
            return self.students[grade_number]
        else:
            return []


    def added(self):
        return self.successfully_added
