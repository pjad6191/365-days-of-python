```python
#Exercism Code:
def primes(limit):
    """
    This function implements the Sieve of Eratosthenes algorithm to find all the prime numbers less than or equal to a given number. 
    Param: int (limit)
    Return: list of primes
    """

    # return an empty list if limit is 1 or less 
    if limit <= 1:
        return []

    # create a list from 2 through limit
    nums = list(range(2,limit+1))

    # loop through to get rid of multiples
    for digit in nums:
        for multiplier in range(2, limit+1):
            product = digit * multiplier
            if product in nums:
                nums.remove(product)
            if product >= limit:
                break
    
    return nums

```
