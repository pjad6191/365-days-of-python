```python
#Exercism Code:
"""
Given any two lists A and B, this function determines if:
List A is equal to list B; or
List A contains list B (A is a superlist of B); or
List A is contained by list B (A is a sublist of B); or
None of the above is true, thus lists A and B are unequal

Param: two lists 
Return: subcategory
"""

# Possible sublist categories.
# Change the values as you see fit.
SUBLIST = 1
SUPERLIST = 2
EQUAL = 3
UNEQUAL = 4


def sublist(list_one, list_two):
    # Convert lists to strings
    str_1 = "," + ",".join(map(str, list_one)) + ","
    str_2 = "," + ",".join(map(str, list_two)) + ","

    # If they are the same, they are equal    
    if list_one == list_two:
        return EQUAL
        
    # If list one is empty, it's a sublist
    if not list_one:   
        return SUBLIST
        
    # If list two is empty list one is a superlist 
    if not list_two:   
        return SUPERLIST

    # Check if list one is in list two 
    if str_2.find(str_1) != -1:
        return SUBLIST

    # Check if list two is in list one
    if str_1.find(str_2) != -1:
        return SUPERLIST

    # If no prior cases, they are unequal
    return UNEQUAL


```
