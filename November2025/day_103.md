```python
#Exercism Code:

class Luhn:
    def __init__(self, card_num):
        self.card_num = card_num

    def valid(self):
        """
        Determine whether a number is valid according to the Luhn formula. The number will be provided as a string.
        param: str
        return: bool
        """
        
        #Validate the number 
        # Remove spaces first
        num = self.card_num.replace(" ", "")

        # Strings of length 1 or less are not valid
        if len(num) <= 1:
            return False

        # All other non-digit characters are disallowed
        if not num.isdigit():
            return False

        # Reverse digits so we can count positions from the right
        digits = []
        for x in num:
            digits.append(int(x))
        digits = digits[::-1]

        # Apply Luhn transformation on every other digit 
        for i in range(1, len(digits), 2):  
            doubled = digits[i] * 2
            if doubled > 9:
                doubled -= 9
            digits[i] = doubled

        # Sum the digits and check if its divisible by 10 
        return sum(digits) % 10 == 0


```
