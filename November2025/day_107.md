```python
def prime(number):
    """
    Given a number n, determine what the nth prime is. 
    """
    #Handle Errors
    if number < 1:
        raise ValueError('there is no zeroth prime')

    #Initialize variables
    count = 0
    num = 2
    nth_num = 2
    
    while count < number:
        is_prime = True
        # Check divisibility up to the square root
        for i in range(2, int(num**0.5) + 1): 
            if num % i == 0:
                is_prime = False
                break
        if is_prime:
            count +=1
            nth_num = num
        num += 1
    return nth_num


```
