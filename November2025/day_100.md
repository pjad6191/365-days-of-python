def say(number):
    """
    Given a number (int), the function returns a string of English words exactly as the number should said out loud. Numbers can range from 0 up to 999,999,999,999
    """
    
    # if the number is negative
    if number < 0:
        raise ValueError("input out of range")

    # if the number is larger than 999,999,999,999
    if number > 999999999999:
        raise ValueError("input out of range")
        
    digits = {"1":"one", 
              "2":"two", 
              "3":"three", 
              "4":"four", 
              "5":"five",
              "6":"six", 
              "7":"seven", 
              "8":"eight",
              "9":"nine"}
    
    teens = {"0":"ten",
             "1":"eleven", 
             "2":"twelve", 
             "3":"thirteen", 
             "4":"fourteen", 
             "5":"fifteen",
             "6":"sixteen", 
             "7":"seventeen", 
             "8":"eighteen",
             "9":"nineteen"}
    
    tens = {"2":"twenty", 
            "3":"thirty", 
            "4":"forty", 
            "5":"fifty", 
            "6":"sixty", 
            "7":"seventy", 
            "8":"eighty", 
            "9":"ninety"}
    
    num = str(number)
    
    if num == "0":
        return "zero"
    
    # Make a working copy for the number 
    workingnum = num

    # Separate the number in chunks
    chunks = []
    while len(workingnum) > 0:
        # Take last 3 digits
        chunks.insert(0, workingnum[-3:]) 
        # Remove last 3 digits
        workingnum = workingnum[:-3]
    
    words_list = []
    
    # Get the words for each chunk 
    for chunk in chunks:
        text = ""
        # Pad with leading zeros so it's always 3 digits
        chunk = chunk.zfill(3)  
        reverse_chunk = chunk[::-1]
        
        for index, char in enumerate(reverse_chunk):
            if index == 0:
                if char == "0":
                    continue
                else:
                    text = digits[char]
            # For teen numbers
            if index == 1 and reverse_chunk[1] == "1":
                text = teens[reverse_chunk[0]]
            # For all numbers 20 and above
            if index == 1 and reverse_chunk[1] != "1" and reverse_chunk[1] != "0":
                if text:
                    text = tens[char] + "-" + text
                else:
                    text = tens[char]
            # For the hundredth place     
            if index == 2 and char != "0":
                text = digits[char] + " hundred " + text
        
        words_list.append(text.strip())

    #Apply place names: billion, million, thousand
    labels = ["billion", "million", "thousand", ""]
    say = ""

    for i in range(len(chunks)):
        if words_list[i]:
            label = labels[-len(chunks) + i]
            say += words_list[i] + " " + label + " "
        
    return say.strip()

def greedy_cow_transport(cows,limit=10):
    """
    Uses a greedy heuristic to determine an allocation of cows that attempts to
    minimize the number of spaceship trips needed to transport all the cows. The
    returned allocation of cows may or may not be optimal.
    The greedy heuristic should follow the following method:

    1. As long as the current trip can fit another cow, add the largest cow that will fit
        to the trip
    2. Once the trip is full, begin a new trip to transport the remaining cows

    Does not mutate the given dictionary of cows.

    Parameters:
    cows - a dictionary of name (string), weight (int) pairs
    limit - weight limit of the spaceship (an int)
    
    Returns:
    A list of lists, with each inner list containing the names of cows
    transported on a particular trip and the overall list containing all the
    trips
    """
    #Create a copy of dictionary so as not to alter original 
    master_list = cows.copy()
    
    #Initialize return list of lists 
    traveling_cow_lists = []


    while len(master_list) > 0:
        #New for each trip
        traveling_cows = []
        total_weight = 0

        #Create a new copy to pop from
        cows2 = master_list.copy()

        while len(cows2) > 0:
            #Get the biggest cow
            biggest = max(cows2, key=cows2.get)

            #Check to see if it will fit
            if total_weight + cows2[biggest] <= 10:
                
                #If it fits
                #Add it to the travel list
                traveling_cows.append(biggest)
                
                #Add the weight to the total weight
                total_weight += cows2[biggest]
                
                #Remove the cow from the list of remaineing cows
                master_list.pop(biggest)
            
            #Remove this cow from temp list because either it fits or its too big
            cows2.pop(biggest)  # always remove from temp list

        #Add this list to the overal list
        traveling_cow_lists.append(traveling_cows.copy())

    return traveling_cow_lists

def brute_force_cow_transport(cows,limit=10):
    """
    Finds the allocation of cows that minimizes the number of spaceship trips
    via brute force.  The brute force algorithm should follow the following method:

    1. Enumerate all possible ways that the cows can be divided into separate trips 
        Use the given get_partitions function in ps1_partition.py to help you!
    2. Select the allocation that minimizes the number of trips without making any trip
        that does not obey the weight limitation
            
    Does not mutate the given dictionary of cows.

    Parameters:
    cows - a dictionary of name (string), weight (int) pairs
    limit - weight limit of the spaceship (an int)
    
    Returns:
    A list of lists, with each inner list containing the names of cows
    transported on a particular trip and the overall list containing all the
    trips
    """
    best_partition = None

    for partition in get_partitions(cows.keys()):
        acceptable_trip = True  
    
        #Check to see if any trip is over the weight limit
        for trip in partition:
            trip_weight = 0
    
            for cow in trip:
                trip_weight += cows[cow]
    
            #If it's over the weight limit, it won't work
            if trip_weight > limit:
                acceptable_trip = False
                break
    
        if acceptable_trip:
            #Only saves the best partition
            if best_partition is None or len(partition) < len(best_partition):
                best_partition = partition 

    return best_partition
