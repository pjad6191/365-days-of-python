```python
def slices(series, length):
    """
    Given a string of digits, output all the contiguous substrings of length n in that string in the order they appear. 
    Param: str
    Return: list of str
    """
    
    # Check for errors:
    if length == 0:
        raise ValueError("slice length cannot be zero")
    
    if length < 0:
        raise ValueError("slice length cannot be negative")
    
    if series == "":
        raise ValueError("series cannot be empty")
    
    if length > len(series):
        raise ValueError("slice length cannot be greater than series length")
    
    substrings = []
    
    # Go from first index up to and including len(series) - length
    for index in range(len(series) - length + 1):
        slice = series[index:index + length] 
        substrings.append(slice)
    
    return substrings


```
