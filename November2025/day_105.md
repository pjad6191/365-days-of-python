```python
#Exercism Code:
class Allergies:

    # Allergen dictionary 
    allergies = {
        0: "eggs",
        1: "peanuts",
        2: "shellfish",
        3: "strawberries",
        4: "tomatoes",
        5: "chocolate",
        6: "pollen",
        7: "cats"
    }

    def __init__(self, score):
        self.score = score

    @property
    def lst(self):
        # Get binary of score
        binary = bin(self.score)  

        # Remove the "0b" at the start
        num = binary[2:]          

        powers = []
        # reverse num so index 0 = rightmost bit
        for index, digit in enumerate(num[::-1]):
            if digit == "1":
                powers.append(index)

        allergic = []
        for num in powers:
            if num in Allergies.allergies:
                allergic.append(Allergies.allergies[num])

        return allergic

    #Checks to see if item is in list of allergens
    def allergic_to(self, item):
        # just check if the item is in the list we already built
        return item in self.lst



```
