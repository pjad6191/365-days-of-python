def saddle_points(matrix):

    # Handle empty matrix
    if not matrix:
        return []

    # Check for irregular matrix
    row_length = len(matrix[0])
    for row in matrix:
        if len(row) != row_length:
            raise ValueError("irregular matrix")

    results = []

    num_rows = len(matrix)
    num_cols = len(matrix[0])

    # Loop through each row
    for r in range(num_rows):
        row = matrix[r]
        largest = max(row)

        # In case of multiple 
        for c in range(num_cols):

            if row[c] == largest:
                column_values = []
                for rr in range(num_rows):
                    column_values.append(matrix[rr][c])

                # Check if it's the smallest in its column
                if row[c] == min(column_values):
                   
                    results.append({
                        "row": r + 1,
                        "column": c + 1
                    })

    return results
