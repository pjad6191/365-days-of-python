#Creates a Dungeons & Dragons character

import random
import math 

class Character:
    def __init__(self):
        self.strength = self.ability()
        self.dexterity = self.ability()
        self.constitution = self.ability()
        self.intelligence = self.ability()
        self.wisdom = self.ability()
        self.charisma = self.ability()
        
        self.hitpoints = 10 + modifier(self.constitution)
              
    def ability(self):
        """
        This is a random character generator that returns a score for an ability.
        """
        rolls = []
        for roll in range(4):
            rolls.append(random.randint(1, 6))
        rolls.remove(min(rolls))
        return sum(rolls)


def modifier(value):
        """
        This function modifies a character's constitution to help determine hit points.
        """
        return math.floor((value - 10)/2)
