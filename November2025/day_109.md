import random
import string

class Robot:
    used_names = set()
    def __init__(self):
        self.name = self.create_unique_name()

    def create_name(self):
        """Creates a unique name"""
        first_letter = random.choice(string.ascii_uppercase)
        second_letter = random.choice(string.ascii_uppercase)
        num = random.randint(100, 999)
        return first_letter + second_letter + str(num)
        
    
    def create_unique_name(self):
        """Creates a name that is unique and not already used."""
        name = self.create_name()
        while name in Robot.used_names:
            name = self.create_name()
        Robot.used_names.add(name)
        return name

    def reset(self):
        """Resets to factory settings so it gets another new name"""
        self.name = self.create_unique_name()
