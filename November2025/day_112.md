```python
def transpose(text):

    if text == "":
        return ""

    words = text.split("\n")
    longest_word = max(words, key=len)
    word_length = len(longest_word)

    for i in range(len(words)):
        if len(words[i]) < word_length:
            words[i] = words[i].ljust(word_length, ' ')
        
    transposed = ""
    
    for index in range(word_length):
        new_word = ""
        for i in range(len(words)):
            new_word += (words[i][index])

        # Only strip padding when we actually had multiple lines
        if len(words) > 1:
            new_word = new_word.rstrip()

        if index < word_length - 1:
            transposed += new_word + "\n"
        else:
            transposed += new_word
        
    return transposed

```
