def transpose(text):

    if text == "":
        return ""

    # Split into lines (rows)
    words = text.split("\n")

    longest_word = max(words, key=len)
    word_length = len(longest_word)

    # Make each the same length
    for i in range(len(words)):
        if len(words[i]) < word_length:
            words[i] = words[i].ljust(word_length, ' ')

  
    transposed_words = []
    # Strip trailing spaces, but don't make it shorter than the line below 
    last = len(transposed_words) - 1
    min_length = len(transposed_words[last].rstrip())
    transposed_words[last] = transposed_words[last].rstrip()

    for i in range(last - 1, -1, -1):
        stripped_len = len(transposed_words[i].rstrip())
        new_len = max(stripped_len, min_length)
        transposed_words[i] = transposed_words[i][:new_len]
        min_length = new_len


    return "\n".join(transposed_words)
